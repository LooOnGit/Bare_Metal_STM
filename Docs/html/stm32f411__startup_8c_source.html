221/* Reset Handler */
222void Reset_Handler(void)
223{
224 // Calculate the sizes of the .data and .bss sections
225 uint32_t data_mem_size = ((uint32_t)&_edata - (uint32_t)&_sdata)/4;
226 uint32_t bss_mem_size = ((uint32_t)&_ebss - (uint32_t)&_sbss)/4;
227
228 // Initialize pointers to the source and destination of the .data section
229 uint32_t *p_src_mem = (uint32_t *)&_etext;
230 uint32_t *p_dest_mem = (uint32_t *)&_sdata;
231
232 /*Copy .data section from FLASH to SRAM*/
233 for(uint32_t i = 0; i < data_mem_size; i++ )
234 {
235
236 *p_dest_mem++ = *p_src_mem++;
237 }
238
239 // Initialize the .bss section to zero in SRAM
240 p_dest_mem = (uint32_t *)&_sbss;
241
242 for(uint32_t i = 0; i < bss_mem_size; i++)
243 {
244 /*Set bss section to zero*/
245 *p_dest_mem++ = 0;
246 }
247
248 // Call the application's main function.
249
250 main();
251}
